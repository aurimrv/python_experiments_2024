class Sort:
    
    @staticmethod
    def selection_sort(arr):''' Selection Sort
            Complexity: O(n^2)
        '''
        for i in range(len(arr)):
            minimum = i
            for j in range(i + 1, len(arr)):
                
                if arr[j] < arr[minimum]:
                    minimum = j
            
            (arr[minimum], arr[i]) = (arr[i], arr[minimum])
        return arr

    def insertion_sort(arr):''' Insertion Sort
            Complexity: O(n^2)

            pos = i

                arr[pos] = arr[pos - 1]
                pos = pos - 1
