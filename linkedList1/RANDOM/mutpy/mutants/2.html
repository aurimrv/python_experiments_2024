        self.head = None
        self.size = 0

    def __len__(self):
        return self.size

    def __contains__(self, item):
        if self.head is not None:
            cur = self.head
            while cur is not None:
                if cur.data is item:
                    return True
                cur = cur.next
        return False
    
    ''' Both getitem and setitem represent the magic methods
        for the object[index] and object[index] = other operations
        for linked lists they run in O(n) time making them less
        efficient than a list() for lookups
    '''
    def __getitem__(self, index):
        if (index >= self.size or index < 0):
            raise IndexError()
        else:
            cur = self.head
            for x in range(index):
                cur = cur.next
            return cur.data
    
    def __setitem__(self, index, value):
        if (index >= self.size or index < 0):
            raise IndexError()
        else:
            cur = self.head
            for x in range(index):
                cur = cur.next
            cur.data = value

            raise StopIteration()

            node = self.cursor.data
            self.cursor = self.cursor.next
