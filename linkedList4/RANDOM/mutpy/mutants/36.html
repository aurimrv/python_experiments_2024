class LinkedList(object):

    push(val) - will insert the value at the head of the list.
    pop() - remove the first value off the head and return it.

    present, else None

    if it were a Python tuple literal: "(12, \'sam\', 37, \'tango\')"
    '''

            if data:
                self.push(data)
